iceflower.alert 弹窗

@Params 传入的参数是一个对象 Object
{
    text: '提示文字',
    confirmBtnText: '按钮文字'
    cancelBtnText: '按钮文字'
    callback: 回调函数,
    width: 不传参的话，默认72vw,
    color: 默认蓝色，你可以传入任意的颜色
    zIndex: 默认3100，当你需要更高的时候可以调高一点, number
    type: 'ios' 或者不填，填了瞬间逼格高一倍
    colorType: 'danger', 'success', 'warning'， 文字控制常用的三种颜色
    cancel: false, true,
    cancelCallback: fn,
    close: false, true   //是否可以手动关闭
}
iceflower.alert()

iceflower.alert({ text: '我是小可爱', colorType: 'danger' })

iceflower.alert({ text: '我是小可爱', type: 'ios', callback: function() { alert('这是一个回调函数'); } })

iceflower.alert({ text: '我是小可爱', callback: function() { alert('这是一个回调函数'); }, cancel: true })

iceflower.alert({ text: '我是小可爱', close: true })

iceflower.alert({ input: true, cancel: true, inputType: 'text', inputPlaceholder: '随便写吧', colorType: 'warning', callback: function(inputValue) { console.log(inputValue); } })

iceflower.toast 小提示

弹出一个Toast
默认2秒自动消失，点击也会消失
@Params is a Object 
{
    text:   '提示文字', 可插入html字符串，例如<br>自主控制换行,
    bgColor: 默认透明黑，你可以传入任意的颜色,
    zIndex:  默认3100，当你需要更高的时候可以调高一点 number,
    vanish:  默认2000毫秒消失
}
iceflower.toast();

iceflower.toast({text: '🤣🤣🤣<br>我是谁，我在哪鸭？', bgColor: 'rgba(15,174,105,0.3)'});

iceflower.loading 加载

弹出一个loading
@Params is a Object 
{
    text: '提示文字',
    color: 默认白色，你可以传入任意的颜色
    zIndex: 默认3100，当你需要更高的时候可以调高一点 number,
    type: 'white' //白色背景的loading
}
iceflower.loading()

iceflower.loading({type: 'white'})

iceflower.loading({size: 'little'})

iceflower.closeLoading 关闭加载

关闭所有loading
不用传参
iceflower.closeloading()

工具和验证

快捷帮元素添加css样式
var el = document.getElementById('flower');
iceflower.addStyle(el, {
    width: '100px',
    height: '200px'
});
    
动态添加css代码到样式表
iceflower.insertCSS('.demo { color: orange; animation: lalaMove }');
    
得到一个两数之间的随机整数，包括两个数在内
var num = iceflower.getRandomNum(1,29);
    
获取页面传参http://www.demo.com?name=cheer&id=290的值
var name = iceflower.urlQuery('name');    //cheer
    
判断是pc端浏览器还是移动端浏览器
返回 1 pc端
返回 0 移动端
var bool = iceflower.checkBrowser();
    
判断手机号码是否正确
只要开头为1且11位就ok
对返回1
错返回0
var bool = iceflower.checkMobile('18000000000');
